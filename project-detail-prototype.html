Design Interrogation

This project is a design investigation into the relationship between time pressure, movement behaviour, and route optimisation within a system defined entirely by traversal mechanics. Rather than generating challenge through combat or enemy encounters, the prototype isolates movement and temporal constraint as the primary forces shaping player experience.

The central hypothesis asks: how does a countdown timer transform movement mechanics from functional navigation tools into systems of strategic decision-making and performance optimisation?

The Challenge

Design a game whose entire challenge emerges from the interaction between a layered movement system and a countdown timer — with no enemies, no combat, and no narrative obstacles. The player must collect a required number of crystals before the timer expires, forcing them to weigh movement speed against precision across a three-level progression. A player who moves slowly may collect all crystals but fail on time. A player who moves recklessly may waste time recovering from errors.

The Solution

The movement system layers six mechanics — horizontal movement, jumping, double-jumping, wall sliding, wall jumping, and dashing — whose individual value is determined entirely by the timer. This transforms traversal from navigation into strategy. The examination build expanded the communicative layer with a full audio system, per-second countdown warnings, a globally managed pause that freezes physics and timer simultaneously, and event-driven architecture for crystal collection and level transitions.

Technical Implementation

Player movement controller — six-mechanic layered system: horizontal movement, jump, double-jump, wall slide, wall jump, and dash. 

// PlayerMovement.cs
using System.Collections;
using UnityEngine;

public class PlayerMovement : MonoBehaviour
{
    [Header("Movement")]
    [SerializeField] private float moveSpeed     = 8f;
    [SerializeField] private float jumpForce     = 14f;
    [SerializeField] private float dashForce     = 20f;
    [SerializeField] private float dashDuration  = 0.15f;
    [SerializeField] private float wallSlideSpeed = 2f;

    [Header("Detection")]
    [SerializeField] private LayerMask groundLayer;
    [SerializeField] private LayerMask wallLayer;
    [SerializeField] private Transform groundCheck;
    [SerializeField] private Transform wallCheck;

    private Rigidbody2D rb;
    private bool isGrounded, isTouchingWall, isWallSliding;
    private int  jumpsRemaining = 2;   // 2 = double-jump allowed
    private bool canDash  = true;
    private bool isDashing = false;

    void Start() => rb = GetComponent<Rigidbody2D>();

    void Update()
    {
        if (isDashing) return;
        CheckCollisions();
        HandleMovement();
        HandleJump();
        HandleWallSlide();
        HandleDash();
    }

    void CheckCollisions()
    {
        isGrounded     = Physics2D.OverlapCircle(groundCheck.position, 0.1f, groundLayer);
        isTouchingWall = Physics2D.OverlapCircle(wallCheck.position,   0.1f, wallLayer);
        if (isGrounded) { jumpsRemaining = 2; canDash = true; }
    }

    void HandleMovement()
    {
        float x = Input.GetAxisRaw("Horizontal");
        rb.velocity = new Vector2(x * moveSpeed, rb.velocity.y);
    }

    void HandleJump()
    {
        if (Input.GetButtonDown("Jump") && isWallSliding)
        {
            // Wall jump: push away from wall
            float dir = isTouchingWall ? -transform.localScale.x : transform.localScale.x;
            rb.velocity = new Vector2(dir * moveSpeed, jumpForce);
            return;
        }
        if (Input.GetButtonDown("Jump") && jumpsRemaining > 0)
        {
            rb.velocity = new Vector2(rb.velocity.x, jumpForce);
            jumpsRemaining--;
        }
    }

    void HandleWallSlide()
    {
        isWallSliding = isTouchingWall && !isGrounded && rb.velocity.y < 0;
        if (isWallSliding)
            rb.velocity = new Vector2(rb.velocity.x, -wallSlideSpeed);
    }

    void HandleDash()
    {
        if (Input.GetKeyDown(KeyCode.LeftShift) && canDash)
            StartCoroutine(Dash());
    }

    IEnumerator Dash()
    {
        isDashing = true;
        canDash   = false;
        float dir = transform.localScale.x;
        rb.velocity     = new Vector2(dir * dashForce, 0f);
        rb.gravityScale = 0f;
        yield return new WaitForSeconds(dashDuration);
        rb.gravityScale = 3f;
        isDashing = false;
    }
}

Countdown timer — per-second audio warning in the final 10 seconds, UnityEvent hook for expiry, and pause/resume control used by the global pause manager. 

// CountdownTimer.cs
using System;
using UnityEngine;
using UnityEngine.Events;

public class CountdownTimer : MonoBehaviour
{
    [SerializeField] private float     startTime    = 60f;
    [SerializeField] private AudioClip warningBeep;

    public UnityEvent onTimerExpired;
    public static event Action<float> OnTimeChanged;

    private float       currentTime;
    private bool        isRunning = true;
    private AudioSource audioSource;
    private int         lastSecond;

    void Start()
    {
        currentTime  = startTime;
        lastSecond   = Mathf.CeilToInt(currentTime);
        audioSource  = GetComponent<AudioSource>();
    }

    void Update()
    {
        if (!isRunning) return;

        currentTime -= Time.deltaTime;
        OnTimeChanged?.Invoke(currentTime);

        // Per-second beep in final 10 seconds
        int currentSecond = Mathf.CeilToInt(currentTime);
        if (currentTime <= 10f && currentSecond != lastSecond)
        {
            audioSource.PlayOneShot(warningBeep);
            lastSecond = currentSecond;
        }

        if (currentTime <= 0f)
        {
            currentTime = 0f;
            isRunning   = false;
            onTimerExpired.Invoke();
        }
    }

    public void Pause()  => isRunning = false;
    public void Resume() => isRunning = true;
}
