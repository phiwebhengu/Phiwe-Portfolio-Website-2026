The Challenge

As a class assignment, our group had to design and ship a complete, polished game with a narrative core, real-time combat, and puzzle mechanics — all within a fixed academic timeline. The brief required a cohesive game world with distinct zones, enemy types, and a story-driven progression system that would feel meaningful to a player rather than mechanical.

The Solution

We built a top-down hack-and-slash using Unity's 2D tools. As Game Developer and Technical Artist, I engineered the combat system, enemy hit detection, and the collectible soul fragment mechanic that drives the core gameplay loop. The game uses a state machine for enemy AI so each boss servant behaves distinctly, giving each encounter a unique tactical feel while keeping the codebase maintainable.

The Story

Two siblings — Otto and Ruri — get transported into a supernatural world. Shortly after arriving, Otto is cursed, transformed into a monkey, and has his soul shattered into five pieces scattered across the lands. To turn him back to human, they must journey through unknown territories, battle powerful servants of the bog, solve puzzles, and explore the forgotten world of the Celestial civilization.

Technical Implementation

Core melee combat script — handles attack input, OverlapCircle hit detection, and damage application to enemy health components. 

// CombatSystem.cs
using System.Collections;
using UnityEngine;

public class CombatSystem : MonoBehaviour
{
    [SerializeField] private float attackRange  = 1.5f;
    [SerializeField] private int   attackDamage = 10;
    [SerializeField] private LayerMask enemyLayer;
    [SerializeField] private Transform attackPoint;

    private Animator animator;
    private bool isAttacking = false;

    void Start()
    {
        animator = GetComponent<Animator>();
    }

    void Update()
    {
        if (Input.GetKeyDown(KeyCode.Z) && !isAttacking)
            StartCoroutine(PerformAttack());
    }

    IEnumerator PerformAttack()
    {
        isAttacking = true;
        animator.SetTrigger("Attack");

        // Detect all enemies in melee range
        Collider2D[] hits = Physics2D.OverlapCircleAll(
            attackPoint.position, attackRange, enemyLayer
        );

        foreach (Collider2D hit in hits)
        {
            EnemyHealth enemy = hit.GetComponent<EnemyHealth>();
            if (enemy != null)
                enemy.TakeDamage(attackDamage);
        }

        yield return new WaitForSeconds(0.4f);
        isAttacking = false;
    }

    void OnDrawGizmosSelected()
    {
        if (attackPoint == null) return;
        Gizmos.DrawWireSphere(attackPoint.position, attackRange);
    }
}
